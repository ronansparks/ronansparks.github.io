class Solution {
    func twoSum(_ nums: [Int], _ target: Int) -> [Int] {
        var result: [Int] = []
        var dict: [Int: Int] = [:]

        for (index, num) in nums.enumerated() {
            let difference = target - num
            if dict[difference] == num {
                let pairedIndex = nums.index(of: difference)!
                result.append(pairedIndex)
                result.append(index)
            }
            else {
                dict[num] = target - num
            }
        }

        return result
    }
}

class Solution {
    func isValidSudoku(_ board: [[Character]]) -> Bool {
        var rowSet = Set<Character>()
        var columnSet = Set<Character>()
        for i in 0..<9 {
            rowSet.removeAll()
            columnSet.removeAll()
            for j in 0..<9 {
                // row check
                if board[i][j] != "." {
                    if rowSet.contains(board[i][j]) {
                        return false
                    }
                    rowSet.insert(board[i][j])
                }

                // column check
                if board[j][i] != "." {
                    if columnSet.contains(board[j][i]) {
                        return false
                    }
                    columnSet.insert(board[j][i])
                }

                // block check
                if i % 3 == 0 && j % 3 == 0 {
                    if !checkBlock(board, row: i, column: j) {
                        return false
                    }
                }
            }
        }
        return true
    }

    func checkBlock(_ board: [[Character]], row: Int, column: Int) -> Bool {
        var blockSet = Set<Character>()
        for i in row..<(row + 3) {
            for j in column..<(column + 3) {
                if board[i][j] != "." {
                    if blockSet.contains(board[i][j]) {
                        return false
                    }
                    blockSet.insert(board[i][j])
                }
            }
        }
        return true
    }
}

class Solution {
    func rotate(_ matrix: inout [[Int]]) {
        let n = matrix[0].count 
        var temp: Int
        for i in 0..<n {
            for j in (i + 1)..<n {
                temp = matrix[i][j]
                matrix[i][j] = matrix[j][i]
                matrix[j][i] = temp
            }
        }

        for i in 0..<n {
            for j in 0..<(n / 2) {
                temp = matrix[i][j]
                matrix[i][j] = matrix[i][n-1-j]
                matrix[i][n-1-j] = temp
            }
        }
    }
}
